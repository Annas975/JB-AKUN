/**
 * Mengimpor React dan ReactDOM untuk rendering aplikasi.
 * @module index
 */

import React from 'react';
import ReactDOM from 'react-dom/client';
import App from './App.jsx';
import './index.css';

/**
 * Membuat root ReactDOM untuk rendering aplikasi React di dalam elemen dengan id 'root'.
 * Menggunakan StrictMode React untuk mode ketat.
 * @function
 * @global
 * @param {string} 'root' - ID elemen tempat aplikasi akan dirender.
 * @returns {void}
 */
ReactDOM.createRoot(document.getElementById('root')).render(
  <React.StrictMode>
    <App />
  </React.StrictMode>
);
